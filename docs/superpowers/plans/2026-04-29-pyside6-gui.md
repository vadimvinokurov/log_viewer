```python
"""Command input widget with Tab autocomplete."""

from __future__ import annotations

from PySide6.QtCore import Signal
from PySide6.QtWidgets import QLineEdit


class CommandInput(QLineEdit):
    """Line edit for entering :commands with Tab autocomplete."""

    command_submitted = Signal(str)

    def __init__(self) -> None:
        super().__init__()
        self.setPlaceholderText(":type command...")
        self._suggestions: list[str] = []
        self._suggestion_index: int = 0

    def keyPressEvent(self, event):
        key = event.key()
        if key == Qt.Key.Key_Tab and self._suggestions:
            self._apply_suggestion()
            return
        if key == Qt.Key.Key_Return:
            text = self.text()
            if text:
                self.command_submitted.emit(text)
                self.clear()
            return
        if key == Qt.Key.Key_Escape:
            self.clear()
            self.setFocus()
            return
        super().keyPressEvent(event)

    def set_suggestions(self, suggestions: list[str]) -> None:
        self._suggestions = suggestions
        self._suggestion_index = 0

    def _apply_suggestion(self) -> None:
        if not self._suggestions:
            return
        self.setText(self._suggestions[self._suggestion_index])
        self._suggestion_index = (self._suggestion_index + 1) % len(self._suggestions)
```

```python
"""Log table: QAbstractTableModel + QTableView with vim-style navigation."""

from __future__ import annotations

from PySide6.QtCore import QAbstractTableModel, QModelIndex, Qt
from PySide6.QtGui import QColor
from PySide6.QtWidgets import QTableView, QVBoxLayout, QWidget

from log_viewer.core.models import LogLine, LogLevel

_COLUMNS = ("Line", "Time", "Category", "Message")

# QColor per level — used for foreground role
_LEVEL_COLORS: dict[str, QColor] = {
    "CRITICAL": QColor(180, 0, 0),
    "ERROR": QColor(200, 50, 50),
    "WARNING": QColor(180, 150, 0),
    "INFO": QColor(60, 60, 60),
    "DEBUG": QColor(0, 140, 140),
    "TRACE": QColor(150, 150, 150),
}


class LogTableModel(QAbstractTableModel):
    """Table model backed by a list of LogLine objects."""

    def __init__(self, lines: list[LogLine] | None = None) -> None:
        super().__init__()
        self._lines: list[LogLine] = lines or []

    def rowCount(self, parent: QModelIndex = QModelIndex()) -> int:
        return len(self._lines)

    def columnCount(self, parent: QModelIndex = QModelIndex()) -> int:
        return len(_COLUMNS)

    def headerData(self, section: int, orientation: Qt.Orientation, role: int = Qt.ItemDataRole.DisplayRole):
        if role != Qt.ItemDataRole.DisplayRole:
            return None
        if orientation == Qt.Orientation.Horizontal and 0 <= section < len(_COLUMNS):
            return _COLUMNS[section]
        return None

    def data(self, index: QModelIndex, role: int = Qt.ItemDataRole.DisplayRole):
        if not index.isValid() or index.row() >= len(self._lines):
            return None
        line = self._lines[index.row()]

        if role == Qt.ItemDataRole.DisplayRole:
            col = index.column()
            if col == 0:
                return str(line.line_number)
            if col == 1:
                return line.time_only
            if col == 2:
                return line.category
            if col == 3:
                return line.message

        if role == Qt.ItemDataRole.ForegroundRole:
            return _LEVEL_COLORS.get(line.level.name)

        if role == Qt.ItemDataRole.UserRole:
            return line

        return None

    def update_lines(self, lines: list[LogLine]) -> None:
        self.beginResetModel()
        self._lines = lines
        self.endResetModel()

    @property
    def lines(self) -> list[LogLine]:
        return self._lines


class LogTableView(QTableView):
    """Log table view with vim-style key navigation."""

    def __init__(self, parent: QWidget | None = None) -> None:
        super().__init__(parent)
        self.setSelectionBehavior(QTableView.SelectionBehavior.SelectRows)
        self.setSelectionMode(QTableView.SelectionMode.SingleSelection)
        self.setAlternatingRowColors(True)
        self.verticalHeader().setVisible(False)
        # Monospace font for the table content
        font = self.font()
        font.setFamily("monospace")
        self.setFont(font)
        self._g_pressed = False
        self._y_pressed = False

    def keyPressEvent(self, event):
        key = event.key()
        text = event.text()

        # Vim: j = down, k = up
        if text == "j":
            self.move_cursor_row(1)
            return
        if text == "k":
            self.move_cursor_row(-1)
            return
        # Ctrl-D = half page down, Ctrl-U = half page up
        if key == Qt.Key.Key_D and event.modifiers() & Qt.KeyboardModifier.ControlModifier:
            self.move_cursor_row(self.height() // self.rowHeight(0) // 2)
            return
        if key == Qt.Key.Key_U and event.modifiers() & Qt.KeyboardModifier.ControlModifier:
            self.move_cursor_row(-(self.height() // self.rowHeight(0) // 2))
            return
        # gg = go to top
        if text == "g":
            if self._g_pressed:
                self._g_pressed = False
                self.scrollToTop()
                self.selectRow(0)
                return
            self._g_pressed = True
            return
        # G = go to bottom
        if text == "G":
            model = self.model()
            if model and model.rowCount() > 0:
                last = model.rowCount() - 1
                self.selectRow(last)
                self.scrollTo(self.model().index(last, 0))
            return
        # yy = copy current line
        if text == "y":
            if self._y_pressed:
                self._y_pressed = False
                self._copy_current_line()
                return
            self._y_pressed = True
            return

        self._g_pressed = False
        self._y_pressed = False
        super().keyPressEvent(event)

    def move_cursor_row(self, delta: int) -> None:
        model = self.model()
        if not model or model.rowCount() == 0:
            return
        current = self.currentIndex().row()
        new_row = max(0, min(current + delta, model.rowCount() - 1))
        self.selectRow(new_row)

    def _copy_current_line(self) -> None:
        index = self.currentIndex()
        model = self.model()
        if not model or not index.isValid():
            return
        line = model.data(index, Qt.ItemDataRole.UserRole)
        if line:
            from PySide6.QtWidgets import QApplication
            QApplication.clipboard().setText(line.raw)
```

```python
"""Category tree widget with checkboxes, search, and 'All' root toggle."""

from __future__ import annotations

from PySide6.QtCore import Qt
from PySide6.QtWidgets import (
    QHBoxLayout,
    QLineEdit,
    QTreeWidget,
    QTreeWidgetItem,
    QVBoxLayout,
    QWidget,
)

from log_viewer.core.models import CategoryNode


class CategoryTreeWidget(QWidget):
    """Categories tab: search field + tree with checkboxes."""

    def __init__(self, parent: QWidget | None = None) -> None:
        super().__init__(parent)
        layout = QVBoxLayout(self)
        layout.setContentsMargins(0, 0, 0, 0)

        self._search = QLineEdit()
        self._search.setPlaceholderText("Search categories...")
        self._search.textChanged.connect(self._on_search)
        layout.addWidget(self._search)

        self._tree = QTreeWidget()
        self._tree.setHeaderHidden(True)
        self._tree.itemChanged.connect(self._on_item_changed)
        layout.addWidget(self._tree)

        self._root: CategoryNode | None = None

    def rebuild(self, root: CategoryNode) -> None:
        """Rebuild the tree from a CategoryNode hierarchy."""
        self._tree.blockSignals(True)
        self._tree.clear()
        self._root = root

        # "All" root node
        all_item = QTreeWidgetItem(self._tree, ["All"])
        all_item.setCheckState(0, Qt.CheckState.Checked)
        all_item.setData(0, Qt.ItemDataRole.UserRole, "__all__")

        for name in sorted(root.children):
            child_node = root.children[name]
            child_item = QTreeWidgetItem(all_item, [f"{name} ({child_node.line_count})"])
            child_item.setCheckState(0, Qt.CheckState.Checked if child_node.enabled else Qt.CheckState.Unchecked)
            child_item.setData(0, Qt.ItemDataRole.UserRole, child_node.full_path)
            self._add_children(child_item, child_node)

        self._tree.expandItem(all_item)
        self._tree.blockSignals(False)

    def _add_children(self, parent_item: QTreeWidgetItem, node: CategoryNode) -> None:
        for name in sorted(node.children):
            child = node.children[name]
            item = QTreeWidgetItem(parent_item, [f"{name} ({child.line_count})"])
            item.setCheckState(0, Qt.CheckState.Checked if child.enabled else Qt.CheckState.Unchecked)
            item.setData(0, Qt.ItemDataRole.UserRole, child.full_path)
            self._add_children(item, child)

    def _on_item_changed(self, item: QTreeWidgetItem, column: int) -> None:
        if column != 0:
            return
        data = item.data(0, Qt.ItemDataRole.UserRole)
        if data == "__all__":
            # Toggle all children
            state = item.checkState(0)
            self._tree.blockSignals(True)
            self._set_all_children(item, state)
            self._tree.blockSignals(False)
        # Propagate up: if all siblings checked, parent checked
        self._update_parent_check(item)

    def _set_all_children(self, parent: QTreeWidgetItem, state: Qt.CheckState) -> None:
        for i in range(parent.childCount()):
            child = parent.child(i)
            child.setCheckState(0, state)
            self._set_all_children(child, state)

    def _update_parent_check(self, item: QTreeWidgetItem) -> None:
        parent = item.parent()
        if parent is None:
            return
        all_checked = all(
            parent.child(i).checkState(0) == Qt.CheckState.Checked
            for i in range(parent.childCount())
        )
        any_checked = any(
            parent.child(i).checkState(0) != Qt.CheckState.Unchecked
            for i in range(parent.childCount())
        )
        self._tree.blockSignals(True)
        if all_checked:
            parent.setCheckState(0, Qt.CheckState.Checked)
        elif any_checked:
            parent.setCheckState(0, Qt.CheckState.PartiallyChecked)
        else:
            parent.setCheckState(0, Qt.CheckState.Unchecked)
        self._tree.blockSignals(False)

    def _on_search(self, text: str) -> None:
        filter_text = text.lower()
        all_item = self._tree.topLevelItem(0)
        if all_item is None:
            return
        self._filter_item(all_item, filter_text)
        self._tree.expandAll()

    def _filter_item(self, item: QTreeWidgetItem, text: str) -> bool:
        if not text:
            item.setHidden(False)
            for i in range(item.childCount()):
                self._filter_item(item.child(i), text)
            return True
        # Show if any child matches
        child_match = False
        for i in range(item.childCount()):
            if self._filter_item(item.child(i), text):
                child_match = True
        self_match = text in item.text(0).lower()
        item.setHidden(not (self_match or child_match))
        return self_match or child_match

    def set_filter(self, text: str) -> None:
        self._search.setText(text)

    def get_disabled_paths(self) -> list[str]:
        """Return full_path for every unchecked node."""
        result: list[str] = []
        all_item = self._tree.topLevelItem(0)
        if all_item is None:
            return result
        self._collect_disabled(all_item, result)
        return result

    def _collect_disabled(self, item: QTreeWidgetItem, result: list[str]) -> None:
        for i in range(item.childCount()):
            child = item.child(i)
            if child.checkState(0) == Qt.CheckState.Unchecked:
                path = child.data(0, Qt.ItemDataRole.UserRole)
                if path and path != "__all__":
                    result.append(path)
            self._collect_disabled(child, result)
```

```python
import sys

    Filter,

class _FileLoadWorker(QThread):
    """Load file lines in a background thread."""
    loaded = Signal(list, str)
    error = Signal(str)

    def __init__(self, path: str) -> None:
        super().__init__()
        self._path = path

    def run(self) -> None:
        try:
            if self._path.startswith(("http://", "https://")):
                import urllib.request
                req = urllib.request.Request(self._path)
                with urllib.request.urlopen(req) as resp:
                    if resp.status >= 400:
                        self.error.emit(f"HTTP {resp.status}: {resp.reason}")
                        return
                    data = resp.read().decode("utf-8", errors="replace")
                    self.loaded.emit(data.splitlines(), self._path)
                    return
            file_path = Path(os.path.expanduser(self._path))
            if not file_path.exists():
                self.error.emit(f"File not found: {self._path}")
                return
            lines = file_path.read_text(encoding="utf-8", errors="replace").splitlines()
            self.loaded.emit(lines, self._path)
        except Exception as e:
            self.error.emit(str(e))

        self._splitter.addWidget(self.side_panel)
        self._splitter.setStretchFactor(0, 3)
        self._splitter.setStretchFactor(1, 1)

        central = QWidget()
        vbox = QHBoxLayout(central)
        vbox.setContentsMargins(0, 0, 0, 0)
        vbox.setSpacing(0)

        self._load_worker.loaded.connect(self._on_file_loaded)
        self._load_worker.error.connect(self._on_file_error)

    def _on_file_loaded(self, lines: list[str], path: str) -> None:
        self.log_store.load_lines(lines, file_path=path)
        self._refresh_display()

        self.bottom_bar.set_status(f"Error: {msg}")

    def _on_command_submitted(self, raw: str) -> None:
        if raw.startswith(":"):
            self._handle_command(raw[1:].strip())
        elif raw.startswith("/"):
            self._do_search(raw[1:], SearchMode.PLAIN, SearchDirection.FORWARD)
        elif raw.startswith("?"):
            self._do_search(raw[1:], SearchMode.PLAIN, SearchDirection.BACKWARD)
        self.log_table.setFocus()

    def _handle_command(self, cmd: str) -> None:
        """Parse and execute a command — mirrors TUI LogViewerApp._handle_command."""
        if not cmd:
            return
        try:
            parsed = parse_command(cmd)
        except ParseError as e:
            self.bottom_bar.set_status(f"Error: {e}")
            return

        name = parsed.name

        if name == "q":
            self.close()
        elif name == "open":
            self._open_file(parsed.text)
        elif name in ("f", "fr", "fs"):
            mode = {"f": SearchMode.PLAIN, "fr": SearchMode.REGEX, "fs": SearchMode.SIMPLE}[name]
            self.log_store.add_filter(Filter(pattern=parsed.text, mode=mode))
            self._refresh_display()
        elif name == "rmf":
            if not parsed.text:
                self.log_store.clear_filters()
            else:
                self.log_store.remove_filter(parsed.text)
            self._refresh_display()
        elif name == "lsf":
            self._toggle_side_panel()
            self.side_panel.setCurrentIndex(1)
        elif name in ("s", "sr", "ss"):
            mode = {"s": SearchMode.PLAIN, "sr": SearchMode.REGEX, "ss": SearchMode.SIMPLE}[name]
            self._do_search(parsed.text, mode, SearchDirection.FORWARD)
        elif name == "cate":
            if not parsed.text:
                self.log_store.enable_all_categories()
            else:
                self.log_store.enable_category(parsed.text)
            self._refresh_display()
        elif name == "catd":
            if not parsed.text:
                self.log_store.disable_all_categories()
            else:
                self.log_store.disable_category(parsed.text)
            self._refresh_display()
        elif name == "lscat":
            self._toggle_side_panel()
        elif name in ("h", "hr", "hs"):
            mode = {"h": SearchMode.PLAIN, "hr": SearchMode.REGEX, "hs": SearchMode.SIMPLE}[name]
            color = parsed.flags.get("color", "red")
            self.log_store.add_highlight(Highlight(pattern=parsed.text, mode=mode, color=color))
            self._refresh_display()
        elif name == "rmh":
            if not parsed.text:
                self.log_store.clear_highlights()
            else:
                self.log_store.highlights = [h for h in self.log_store.highlights if h.pattern != parsed.text]
            self._refresh_display()
        elif name == "lsh":
            self._toggle_side_panel()
            self.side_panel.setCurrentIndex(2)
        elif name == "preset":
            parts = parsed.text.strip().split(None, 1)
            if len(parts) == 2:
                action, pname = parts
                if action == "save":
                    self._save_preset(pname)
                elif action == "load":
                    self._load_preset(pname)
            self._refresh_display()
        elif name == "rmpreset":
            try:
                self._presets.delete(parsed.text)
            except FileNotFoundError:
                self.bottom_bar.set_status(f"Preset not found: {parsed.text}")
        elif name == "lspreset":
            names = self._presets.list_presets()
            self.bottom_bar.set_status("Presets: " + ", ".join(names) if names else "No presets")
        elif name == "reload":
            if self.log_store.current_file:
                self._open_file(self.log_store.current_file)
            else:
                self.bottom_bar.set_status("No file loaded")

    def _do_search(self, pattern: str, mode: SearchMode, direction: SearchDirection) -> None:
        if not pattern:
            return
        self.log_store.search(pattern, mode, direction=direction)
        self._update_status()
        self._jump_to_search_match()

    def _save_preset(self, name: str) -> None:
        disabled = [
            path for path, node in self._iter_category_nodes() if not node.enabled
        ]
        state = {
            "filters": self.log_store.filters,
            "highlights": self.log_store.highlights,
            "disabled_categories": disabled,
        }
        self._presets.save(name, state)

    def _load_preset(self, name: str) -> None:
        try:
            data = self._presets.load(name)
        except FileNotFoundError:
            self.bottom_bar.set_status(f"Preset not found: {name}")
            return
        # Apply filters
        existing = {(f.pattern, f.mode.value) for f in self.log_store.filters}
        for fd in data.get("filters", []):
            if (fd["pattern"], fd["mode"]) not in existing:
                self.log_store.add_filter(Filter(fd["pattern"], SearchMode(fd["mode"])))
        # Apply highlights
        existing_hl = {(h.pattern, h.color) for h in self.log_store.highlights}
        for hd in data.get("highlights", []):
            if (hd["pattern"], hd.get("color", "red")) not in existing_hl:
                self.log_store.add_highlight(
                    Highlight(hd["pattern"], SearchMode(hd["mode"]), color=hd.get("color", "red"))
                )
        for cat in data.get("disabled_categories", []):
            self.log_store.disable_category(cat)

    def _iter_category_nodes(self):
        def _walk(node):
            for child in node.children.values():
                yield child.full_path, child
                yield from _walk(child)
        yield from _walk(self.log_store.category_tree)

    def _refresh_display(self) -> None:
        """Refresh log table, side panel, and status bar from LogStore."""
        store = self.log_store
        visible_lines = [store.lines[i] for i in store.filtered_indices]
        self._table_model.update_lines(visible_lines)
        self._refresh_side_panel()
        self._update_status()

    def _refresh_side_panel(self) -> None:
        store = self.log_store
        self.side_panel.category_tree.rebuild(store.category_tree)
        self.side_panel.filter_list.set_filters(store.filters)
        self.side_panel.highlight_list.set_highlights(store.highlights)

    def _update_status(self) -> None:
        store = self.log_store
        filename = store.current_file or "no file"
        visible = len(store.filtered_indices)
        total = len(store.lines)
        parts = [f"{filename}", f"{visible}/{total} lines"]
        # Level counts
        for level, count in store.visible_level_counts.items():
            if count:
                parts.append(f"{level.icon_plain}{count}")
        # Search match info
        if store.search_state and store.search_state.matches:
            ss = store.search_state
            parts.append(f"match {ss.current_index + 1}/{len(ss.matches)}")
        self.bottom_bar.set_status(" | ".join(parts))

    def _jump_to_search_match(self) -> None:
        ss = self.log_store.search_state
        if not ss or not ss.matches:
            return
        matched_idx = ss.matches[ss.current_index]
        if matched_idx in self.log_store.filtered_indices:
            row = self.log_store.filtered_indices.index(matched_idx)
            self.log_table.selectRow(row)
            self.log_table.scrollTo(self._table_model.index(row, 0))

    # Key handling for the main window
    def keyPressEvent(self, event):
        key = event.key()
        text = event.text()

        if self.log_table.hasFocus():
            if text == ":":
                self.bottom_bar.activate_command_mode()
                return
            if text == "/":
                self.bottom_bar.command_input.setText("/")
                self.bottom_bar.command_input.setFocus()
                return
            if text == "n" and self.log_store.search_state:
                self.log_store.next_match()
                self._update_status()
                self._jump_to_search_match()
                return
            if text == "N" and self.log_store.search_state:
                self.log_store.prev_match()
                self._update_status()
                self._jump_to_search_match()
                return
        super().keyPressEvent(event)

    # Drag & Drop
    def dragEnterEvent(self, event):
        if event.mimeData().hasUrls():
            event.acceptProposedAction()

    def dropEvent(self, event):
        for url in event.mimeData().urls():
            path = url.toLocalFile()
            if path:
                self._open_file(path)
                break  # Only first file


def main() -> None:
    app = QApplication(sys.argv)
    file_path = sys.argv[1] if len(sys.argv) > 1 else None
    window = MainWindow(file_path=file_path)
    window.show()
    sys.exit(app.exec())
```
